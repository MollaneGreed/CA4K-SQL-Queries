ALTER PROCEDURE [xcpt].[ExceptionNotification]
    @RECIPIENTS NVARCHAR(255)
  , @JOBNAME NVARCHAR(1024)
AS
    DECLARE @TransactionDTTM VARCHAR(30)
          , @TransactionMark VARCHAR(32)
          , @ErrorMessage NVARCHAR(MAX);

    SET NOCOUNT ON;
    SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
    SET XACT_ABORT ON;

    SET @TransactionDTTM = ( CONVERT(VARCHAR(20), GETDATE(), 112) + '_' + REPLACE(REPLACE(CONVERT(VARCHAR(20), GETDATE(), 114), ':', ''), '.', ''));
    SET @TransactionMark = 'xcptPersonBadge_' + @TransactionDTTM;

    BEGIN TRY
        BEGIN TRANSACTION @TransactionMark WITH MARK;

        DECLARE @PROFILE_NAME NVARCHAR(1024);
        DECLARE @SUBJECT NVARCHAR(1024);
        DECLARE @BODY NVARCHAR(1024);
        DECLARE @HTML_SUBJECT NVARCHAR(MAX);
        DECLARE @HTML NVARCHAR(MAX);
        DECLARE @QUERYSQL NVARCHAR(MAX);

        SET @PROFILE_NAME = N'ADMIN';

        SET @SUBJECT = @JOBNAME + N' - Personnel/Badge records need attention';

        SET @HTML_SUBJECT = N'<p>Personnel/Badge records that need attention</p>';
        SET @HTML_SUBJECT = @HTML_SUBJECT + N'<p>Attachment list those Continental personnel and or badge records that need attention.</p>';

        SET @HTML = @HTML_SUBJECT + N'<p>Notification generated from job ' + @JOBNAME + N' on server ' + @@ServerName + N'</p>';

        TRUNCATE TABLE [xcpt].[PersonBadge];

        --multiple ssn
        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT     1
                             , [bdg].[Badge]
                             , [bdg].[Facility]
                             , TRY_CONVERT(INT, [per].[SSN])
                             , [per].[FrstName]
                             , [per].[LastName]
                             , ''
                             , ''
                             , 'Duplicate Personnel on SSN.  Badges have been disabled.'
                             , 'Not allow. Delete the incorrect personnel and badge record, ensure UMS employeeID is correct and re-enable badge.'
                    FROM       [stg].[caPerson] [per]
                    INNER JOIN [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    WHERE      [per].[SSN] IN (
                                                  SELECT     [per].[SSN]
                                                  FROM       [stg].[caPerson] [per]
                                                  INNER JOIN [stg].[caBadge] [bdg]
                                                      ON [bdg].[PersonID] = [per].[PersonID]
                                                  WHERE      [per].[SSN] IS NOT NULL
                                                  GROUP BY   [per].[SSN]
                                                  HAVING     COUNT(*) > 1
                                              )
                    ORDER BY   [per].[SSN];

        --Multile employeeid
        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT     2
                             , [bdg].[Badge]
                             , [bdg].[Facility]
                             , TRY_CONVERT(INT, [udf].[UserFieldValue])
                             , [per].[FrstName]
                             , [per].[LastName]
                             , ''
                             , ''
                             , 'Duplicate personnel on EmployeeID.  Badges have been disabled'
                             , 'Not allow. Delete the incorrect personnel and badge record, ensure UMS employeeID is correct and re-enable badge.'
                    FROM       [stg].[caPerson] [per]
                    INNER JOIN [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    INNER JOIN [stg].[caUserFields] [udf]
                        ON [udf].[Badge] = [bdg].[Badge]
                           AND [udf].[Facility] = [bdg].[Facility]
                           AND [udf].[PersonID] = [bdg].[PersonID]
                           AND [udf].[UserFieldID] = (
                                                         SELECT [UserFieldID]
                                                         FROM   [stg].[caUserFieldDef]
                                                         WHERE  [UserFieldName] = 'EmployeeID:'
                                                     )
                    WHERE      [udf].[UserFieldValue] IN (
                                                             SELECT     [udf].[UserFieldValue]
                                                             FROM       [stg].[caBadge] [bdg]
                                                             INNER JOIN [stg].[caUserFields] [udf]
                                                                 ON [udf].[Badge] = [bdg].[Badge]
                                                                    AND [udf].[Facility] = [bdg].[Facility]
                                                                    AND [udf].[PersonID] = [bdg].[PersonID]
                                                                    AND [udf].[UserFieldID] = (
                                                                                                  SELECT [UserFieldID]
                                                                                                  FROM   [stg].[caUserFieldDef]
                                                                                                  WHERE  [UserFieldName] = 'EmployeeID:'
                                                                                              )
                                                             WHERE      [udf].[UserFieldValue] <> 'N/A'
                                                             GROUP BY   [udf].[UserFieldValue]
                                                             HAVING     COUNT(*) > 1
                                                         );


        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    --invalid
                    SELECT          3
                                  , [bdg].[Badge]
                                  , [bdg].[Facility]
                                  , 0
                                  , [per].[FrstName]
                                  , [per].[LastName]
                                  , COALESCE([emp].[HRSiteCode], '')
                                  , COALESCE([emp].[HRSiteName], '')
                                  , 'Invalid.  Can not relate badge to an employee or designated non-employee situation.  Access removed and badge disabled.'
                                  , 'Not allowed.  Delete the badge records or ensure correct setup if non-employee. If HRSite listed then that site is not part of the automation configuration.'
                    FROM            [stg].[caPerson] [per]
                    INNER JOIN      [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    LEFT OUTER JOIN [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, [per].[SSN])
                    WHERE           [per].[LastName] LIKE '%invalid%'
                                    AND COALESCE([per].[CompanyID], 0) = 0;



        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT          4
                                  , [bdg].[Badge]
                                  , [bdg].[Facility]
                                  , 0
                                  , [per].[FrstName]
                                  , [per].[LastName]
                                  , [emp].[HRSiteCode]
                                  , [emp].[HRSiteName]
                                  , 'Enabled Temp or Exec badge unable to related to employee.  Badge disabled'
                                  , 'Update personnel record Last Name and ensure it is prefixed with the employee id followed by a comma, Example: "0000000, Name"'
                    FROM            [stg].[caPerson] [per]
                    INNER JOIN      [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    INNER JOIN      [map].[AutomationExternalExclusion] [ext]
                        ON [ext].[ID] = [per].[CompanyID]
                           AND [ext].[IsEmployee] = 1
                    LEFT OUTER JOIN [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, REPLACE(SUBSTRING([per].[LastName], 1, CHARINDEX(',', [per].[LastName])), ',', ''))
                    WHERE           [emp].[EmployeeID] IS NULL
                                    AND [bdg].[Enabled] <> 0;

        --Out of band employee terminated
        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT     5
                             , [bdg].[Badge]
                             , [bdg].[Facility]
                             , TRY_CONVERT(INT, REPLACE(SUBSTRING([per].[LastName], 1, CHARINDEX(',', [per].[LastName])), ',', ''))
                             , [per].[FrstName]
                             , [per].[LastName]
                             , [emp].[HRSiteCode]
                             , [emp].[HRSiteName]
                             , 'Temp or Exec badge assocated with a terminated employee.  Access removed and badge disabled'
                             , 'Remove employee Id from last name on personnel record or delete from the system'
                    FROM       [stg].[caPerson] [per]
                    INNER JOIN [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    INNER JOIN [map].[AutomationExternalExclusion] [ext]
                        ON [ext].[ID] = [per].[CompanyID]
                           AND [ext].[IsEmployee] = 1
                    INNER JOIN [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, REPLACE(SUBSTRING([per].[LastName], 1, CHARINDEX(',', [per].[LastName])), ',', ''))
                           AND [emp].[TerminationDate] IS NOT NULL;

        --active no access groups
        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT          6
                                  , [bdg].[Badge]
                                  , [bdg].[Facility]
                                  , [emp].[EmployeeID]
                                  , [per].[FrstName]
                                  , [per].[LastName]
                                  , [emp].[HRSiteCode]
                                  , [emp].[HRSiteName]
                                  , 'Active Employee and badge has no access groups.'
                                  , 'Check with ID Team and IMAC role and ensure correctly setup. Role: ' + [emprl].[RoleName]
                    FROM            [stg].[caPerson] [per]
                    INNER JOIN      [stg].[ValidEmployee] [ve]
                    INNER           JOIN      [stg].[rbacEmployeeRole] [emprl]
                        ON [emprl].[EmployeeID] = [ve].[EmployeeID]
                    INNER           JOIN      [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = [ve].[EmployeeID]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, [per].[SSN])
                           AND [emp].[TerminationDate] IS NULL
                    INNER JOIN      [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    LEFT OUTER JOIN [CardAccessLiveConfigurationPH].[dbo].[BadgeAccess] [bdgacc]
                        ON [bdgacc].[Badge] = [bdg].[Badge]
                           AND [bdgacc].[Facility] = [bdg].[Facility]
                    WHERE           
									[bdgacc].[Badge] IS NULL
                                    AND [bdg].[Badge] < 1000000000
									AND emp.EmployeeStatusCode <> 'L'; --wvvjr SC69649



        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT     7
                             , [bdg].[Badge]
                             , [bdg].[Facility]
                             , TRY_CONVERT(INT, [per].[SSN])
                             , [per].[FrstName]
                             , [per].[LastName]
                             , [emp].[HRSiteCode]
                             , [emp].[HRSiteName]
                             , 'Employee is part of more that 16 access groups. System Limit, only first 16 applied'
                             , 'Check with ID team and role setup in IMAC as employees can not be part of more than 16 continental access groups'
                    FROM       [stg].[caPerson] [per]
                    INNER JOIN [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    INNER JOIN [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, [per].[SSN])
                    WHERE      TRY_CONVERT(INT, [per].[SSN])IN (
                                                                   SELECT   [grp].[employeeid]
                                                                   FROM     [stg].[ADUserGroup] [grp]
                                                                   WHERE    [grp].[name] LIKE 'AG-CAPHI-%'
                                                                   GROUP BY [grp].[employeeid]
                                                                   HAVING   COUNT(*) > 16
                                                               );

        INSERT INTO [xcpt].[PersonBadge] (
                                             [Seq]
                                           , [Badge]
                                           , [Facility]
                                           , [EmployeeID]
                                           , [FirstName]
                                           , [LastName]
                                           , [HRSiteCode]
                                           , [HRSiteName]
                                           , [Reason]
                                           , [Action]
                                         )
                    SELECT          8
                                  , [bdg].[Badge]
                                  , [bdg].[Facility]
                                  , [emp].[EmployeeID]
                                  , [per].[FrstName]
                                  , [per].[LastName]
                                  , [emp].[HRSiteCode]
                                  , [emp].[HRSiteName]
                                  , 'Active employee badge assigned access group(s) that have no access defined.'
                                  , 'Setup the access group(s) in Continental.  Groups: ' + STRING_AGG([magrp].[Description], ', ')
                    FROM            [stg].[caPerson] [per]
                    INNER JOIN      [dbo].[dimEmployeeAllFutureHireTermV2] [emp]
                        ON [emp].[EmployeeID] = TRY_CONVERT(INT, [per].[SSN])
                           AND [emp].[TerminationDate] IS NULL
                    INNER JOIN      [stg].[caBadge] [bdg]
                        ON [bdg].[PersonID] = [per].[PersonID]
                    INNER JOIN      [CardAccessLiveConfigurationPH].[dbo].[BadgeAccess] [bdgacc]
                        ON [bdgacc].[Badge] = [bdg].[Badge]
                           AND [bdgacc].[Facility] = [bdg].[Facility]
                    INNER JOIN      [CardAccessLiveConfigurationPH].[dbo].[MAccGrp] [magrp]
                        ON [magrp].[AgrpNo] = [bdgacc].[AGroupNo]
                           AND [magrp].[Description] NOT LIKE 'AG-CA%-NoAccess'
                    LEFT OUTER JOIN [CardAccessLiveConfigurationPH].[dbo].[AccGrp] [agrp]
                        ON [agrp].[Agno] = [magrp].[AgrpNo]
                    WHERE           [bdg].[Badge] < 1000000000
                                    AND [agrp].[Agno] IS NULL
                    GROUP BY        [bdg].[Badge]
                                  , [bdg].[Facility]
                                  , [emp].[EmployeeID]
                                  , [per].[FrstName]
                                  , [per].[LastName]
                                  , [emp].[HRSiteCode]
                                  , [emp].[HRSiteName];



        COMMIT TRANSACTION @TransactionMark;

        SET @QUERYSQL = N'SELECT PB.[Badge],PB.[Facility],PB.[EmployeeID],PB.[FirstName],PB.[LastName],PB.[HRSiteCode],PB.[HRSiteName],PB.[Reason],PB.[Action],O.[OperLoginName] AS [Last Operator] FROM [xcpt].[PersonBadge] PB JOIN [stg].[caBadge] B ON PB.[Badge] = B.Badge JOIN [CardAccessLiveConfigurationPH].[dbo].[Operators] O ON B.[LastOperator] = O.OperatorId ORDER BY [Seq] ,[EmployeeID]';

        IF EXISTS (
                      SELECT TOP 1 'X'
                      FROM   [CardAccessCustomPH].[xcpt].[PersonBadge]
                  )
            BEGIN

                EXEC [msdb].[dbo].[sp_send_dbmail] @profile_name = @PROFILE_NAME
                                                 , @recipients = @RECIPIENTS
                                                 , @subject = @SUBJECT
                                                 , @body = @HTML
                                                 , @query = @QUERYSQL
                                                 , @query_result_width = 4000
                                                 , @execute_query_database = 'CardAccessCustomPH'
                                                 , @attach_query_result_as_file = 1
                                                 , @query_attachment_filename = 'ContinentalRecordExceptions.csv'
                                                 , @query_result_separator = '	' --Tab
                                                 , @query_result_no_padding = 1
                                                 , @body_format = 'HTML';
            END;




    END TRY
    BEGIN CATCH
        --	Capture all the error information so it can be included
        --	in the error message and troubleshooting can be more
        --	efficient.

        DECLARE @ERROR_NUMBER INT
              , @ERROR_SEVERITY INT
              , @ERROR_STATE INT
              , @ERROR_PROCEDURE NVARCHAR(128)
              , @ERROR_LINE INT
              , @ERROR_MESSAGE NVARCHAR(4000)
              , @TransactionOutcome CHAR(1);

        SELECT @ERROR_NUMBER = ERROR_NUMBER()
             , @ERROR_SEVERITY = ERROR_SEVERITY()
             , @ERROR_STATE = ERROR_STATE()
             , @ERROR_PROCEDURE = ERROR_PROCEDURE()
             , @ERROR_LINE = ERROR_LINE()
             , @ERROR_MESSAGE = ERROR_MESSAGE();

        --	If the transaction count is greater than zero,
        --	peform a rollback with the transaction mark first,
        --	then perform a rollback if additional transactions
        --	remain.

        --	Test XACT_STATE:  
        --	If 1, the transaction is committable.  
        --	If -1, the transaction is uncommittable and should   
        --	be rolled back.  
        --	If 0, then there is no transaction and  
        --	a commit or rollback operation would generate an error.  

        IF ( XACT_STATE()) = -1
            BEGIN
                ROLLBACK TRANSACTION @TransactionMark;
                SET @TransactionOutcome = 'R';
            END;
        ELSE IF ( XACT_STATE()) = 1
                 BEGIN
                     COMMIT TRANSACTION @TransactionMark;
                     SET @TransactionOutcome = 'C';
                 END;
        ELSE
            SET @TransactionOutcome = '';

        --	The THROW function below will error out if the error number
        --	is below 50000, so add 50000 to the error number if below
        --	50000.
        IF @ERROR_NUMBER < 50000
            SET @ERROR_NUMBER = @ERROR_NUMBER + 50000;

        --	Build the error message that will display the error to the executing process so that
        --	troubleshooting can be performed easily.  Include the stored procedure name and any
        --	parameters that are not confidential, such as passwords or someones SSN.
        --	NOTE:
        --	If you only have one parameter, then you will need to remove the others in the template.
        --	If you have two parameters, then you will need to remove "<next parameter name>" line.
        --	If you have more than three, then you will need to add more "<next parameter name>" lines.

        SET @ErrorMessage = N'An error occured in [xcpt].[InsertPersonBadge]. ' + CHAR(10) + N'Transaction was ' + CASE @TransactionOutcome
                                                                                                                        WHEN 'R' THEN 'Rolledback with transaction mark of "' + @TransactionMark + '".'
                                                                                                                        WHEN 'C' THEN 'Committed with transaction mark of "' + @TransactionMark + '".'
                                                                                                                        ELSE 'No transaction was available for rollback or committed.'
                                                                                                                   END + CHAR(10) + N'Error is:  ' + CHAR(10) + N'ERROR_NUMBER - '
                            + CAST(@ERROR_NUMBER AS NVARCHAR(23)) + CHAR(10) + N'ERROR_SEVERITY - ' + CAST(@ERROR_SEVERITY AS NVARCHAR(23)) + CHAR(10) + N'ERROR_STATE - '
                            + CAST(@ERROR_STATE AS NVARCHAR(23)) + CHAR(10) + N'ERROR_PROCEDURE - ' + @ERROR_PROCEDURE + CHAR(10) + N'ERROR_LINE - ' + CAST(@ERROR_LINE AS NVARCHAR(23)) + CHAR(10)
                            + N'ERROR_MESSAGE - ' + @ERROR_MESSAGE + CHAR(10) + CHAR(10);
        --	After building the error message, throw a new error with the error information.
        THROW @ERROR_NUMBER, @ErrorMessage, @ERROR_STATE;
    END CATCH;
GO
